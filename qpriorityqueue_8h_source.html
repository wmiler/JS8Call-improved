54template <class T, typename LessThan = std::less < T > >
55class QPriorityQueuePrivate {
56public:

70 return (i - 1) / 2;

84template <class T, typename LessThan = std::less < T > >
85class Q_CORE_EXPORT QPriorityQueue
86{
87public:
88 inline QPriorityQueue(LessThan l = std::less < T >())
89 : lessThan(l), d(lessThan) { }

91 : lessThan(q.lessThan), d(q.d) { }
92 inline ~QPriorityQueue() { }
93
94 QPriorityQueue<T> &operator=(const QPriorityQueue<T> &q) {
95 d = q.d; return *this;
96 }
97
98 inline int size() const { return d.size(); }
99 inline bool isEmpty() const { return empty(); }
100
101 // like qlist
102 void clear();
103 void append(const T &t) { return enqueue(t); }
104 void append(const QList<T> &t);
105 T takeFirst() { return dequeue(); }
106 inline int length() const { return size(); } // Same as count()
107 inline T& first() { return head(); }
108 inline const T& first() const { return head(); }
109 inline void removeFirst() { pop(); }
110 inline bool startsWith(const T &t) const
111 { return !isEmpty() && first() == t; }
112
113 // like qqueue
114 inline void enqueue(const T &t) { push(t); }
115 inline T dequeue() { T t = d.top(); d.pop(); return t; }
116 inline T &head() { return const_cast<T&>(top()); }
117 inline const T &head() const { return top(); }

120 typedef int size_type;
121 typedef T value_type;
122
123 inline bool empty() const { return d.empty(); }
124 inline const value_type& top() { Q_ASSERT(!isEmpty()); return d.top(); }
125 inline void push(const value_type& x) { return d.push(x); }
126 inline void pop() { Q_ASSERT(!isEmpty()); d.pop(); }

185// less efficient.
186#ifdef QT_NO_STL
193template <typename T, typename LessThan>
194Q_OUTOFLINE_TEMPLATE void QPriorityQueuePrivate<T, LessThan>::pop()
195{
196 int i = 0;
197 ssize_t size = d.size();;
198
199 if(!size)
200 return;
201 if(size == 1)
202 return d.clear();
203
204 d[0] = d.takeLast();
205
206 while(i < size - 1) {
207 int left = leftChild(i);
208 int right = rightChild(i);
209 bool validLeft = left < size;
210 bool validRight = right < size;
211
212 if(validLeft && lessThan(d.at(i), d.at(left)))
213 if(validRight && !lessThan(d.at(right), d.at(left))) {
214 d.swap(i, right);
215 i = right;
216 } else {
217 d.swap(i, left);
218 i = left;
219 }
220 else if(validRight && lessThan(d.at(i), d.at(right))) {
221 d.swap(i, right);
222 i = right;
223 }
224 else
225 break;
226 }
227}

234template <typename T, typename LessThan>
235Q_OUTOFLINE_TEMPLATE void QPriorityQueuePrivate<T, LessThan>::push(const T &value)
236{
237 int i = d.size();
238 d.append(value);
239 while(i != 0 && !lessThan(d.at(i), d.at(parent(i)))) {
240 d.swap(i, parent(i));
241 i = parent(i);
242 }
243}
