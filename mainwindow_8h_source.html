618 struct FrameCacheKey
619 {
620 int submode;
621 QString frame;
622
623 FrameCacheKey(int submode,
624 QString frame)
625 : submode(submode)
626 , frame (std::move(frame))
627 {}
628
629 bool operator==(FrameCacheKey const &) const noexcept = default;
630
631 struct Hash
632 {
633 std::size_t
634 operator()(FrameCacheKey const & key) const noexcept
635 {
636 std::size_t const h1 = std::hash<int>{}(key.submode);
637 std::size_t const h2 = qHash(key.frame);
638 return h1 ^ (h2 + 0x9e3779b9 + (h1 << 6) + (h1 >> 2));
639 }
640 };
641 };
